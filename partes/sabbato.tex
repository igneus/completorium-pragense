\formulary{Officium sanctae Mariae sabbato}

\rubrica{Nota quod ab Epiphania usque ad Purificationem
  omni sabbato, nisi festum novem lectionum impediat,
  officium sanctae Mariae facimus.}\footcite[110v-111r]{bp1502}

\rubrica{\editorial{Post orationem vesperarum}
  cantantes vero responsorium \incipit{Verbum caro}
  faciunt processionem ad altare Beatae Virginis
  et ibi debito modo vesperae cum completorio finiuntur.}

\rubrica{Ad completorium dicitur ut in prima vespera
  Circumcisionis.\quickref{form:circumcisio}}
