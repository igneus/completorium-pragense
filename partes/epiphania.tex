\formulary{In Epiphania Domini}
\label{form:epiphania}

\rubrica{\editorial{Post primam vesperam}
  ad completorium super psalmos antiphona
  \incipit{Verbum caro}\quickref{can:nativitas_verbumcaro}
  et haec usque ad octavam dicitur.}

\rubrica{Capitulum, hymnus, versus et preces non dicuntur.}

\rubrica{Ad Nunc dimittis antiphona}
Stella ista sicut flamma coruscat
et regem regum demonstrat.
Magi eam viderunt
et Christo regi munera obtulerunt.\quickref{can:epiphania_stella}

\rubrica{Sequitur oratio immediate.}

Deus illuminator omnium gentium,
da populis tuis perpetua pace gaudere:
et illud lumen splendidum infunde cordibus nostris:
quod trium magorum mentibus aspirasti.
Dominum nostrum.

\rubrica{Non plus.}\footcite[114r]{bp1502}

\rubrica{\editorial{Post secundas vesperas}
  ad completorium omnia ut in Nativitate Domini.\quickref{form:nativitas}}

\rubrica{Capitulum}
Omnes de Sabba venient aurum et thus deferentes:
et laudem Domino annuntiantes.
Deo gratias.

\rubrica{Super Nunc dimittis antiphona
  \incipit{Stella ista.}\quickref{can:epiphania_stella}
  Preces dicuntur.}
